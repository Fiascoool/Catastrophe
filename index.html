          <h2>第二组号码</h2>

          <button onclick="generateAge()">选号</button>

        </div>

      <div class="sum">

        <h2>两组数之和</h2>

      </div>

    <script>

      var numbers = [];

      var age = null;


      function generateNumbers() {

        numbers = [];

        while (numbers.length < 6) {

          var n = Math.floor(Math.random() * 33) + 1;

          if (numbers.indexOf(n) === -1) {

            numbers.push(n);

          }

        }

        var result = numbers.join(", ");

        document.querySelector(".numbers:nth-of-type(1) .result").textContent = result;

      }

      

      function generateAge() {

        age = Math.floor(Math.random() * 12) + 1;

        document.querySelector(".numbers:nth-of-type(2) .result").textContent = age;

      }


      function sumNumbers() {

        if (age === null) {

          alert("请先生成第二组数！");

          return;

        }

        var sum = numbers.reduce(function(acc, cur) {

          return acc + cur;

        }, 0) + age;

        document.querySelector(".sum .result").textContent = sum;

      }

    </script>
